  // Limit
  if (transformation.limit) {
    result = result.slice(0, transformation.limit);
  }
  
  return result;
}

// utils/chartValidator.js
export function canCreateChart(chartType, selectedFields, dataSchema) {
  const requirements = {
    bar: {
      categories: { min: 1, max: 2, types: ['string', 'date'] },
      measures: { min: 1, max: 5, types: ['number'] }
    },
    line: {
      categories: { min: 1, max: 1, types: ['date'] },
      measures: { min: 1, max: 10, types: ['number'] }
    },
    pie: {
      categories: { min: 1, max: 1, types: ['string'] },
      measures: { min: 1, max: 1, types: ['number'] }
    },
    table: {
      any: { min: 1, max: 999 }
    }
  };
  
  const req = requirements[chartType];
  const categories = selectedFields.filter(f => 
    req.categories.types.includes(dataSchema[f].type)
  );
  const measures = selectedFields.filter(f =>
    req.measures.types.includes(dataSchema[f].type)
  );
  
  return (
    categories.length >= req.categories.min &&
    categories.length <= req.categories.max &&
    measures.length >= req.measures.min &&
    measures.length <= req.measures.max
  );
}
